17public class SendTextMessage {
18
19    public static final String API_KEY = "your-api-key-goes-here";
20    public static final String API_SECRET = "your-api-secret-goes-here";
21
22    public static final String SMS_FROM = "12345";
23    public static final String SMS_TO = "447777111222";
24    public static final String SMS_TEXT = "Hello World!";
25
26    public static void main(String[] args) {
27
28        // Create a client for submitting to Nexmo
29
30        NexmoSmsClient client = null;
31        try {
32            client = new NexmoSmsClient(API_KEY, API_SECRET);
33        } catch (Exception e) {
34            System.err.println("Failed to instanciate a Nexmo Client");
35            e.printStackTrace();
36            throw new RuntimeException("Failed to instanciate a Nexmo Client");
37        }
38
39        // Create a Text SMS Message request object ...
40
41        TextMessage message = new TextMessage(SMS_FROM, SMS_TO, SMS_TEXT);
42
43        // Use the Nexmo client to submit the Text Message ...
44
45        SmsSubmissionResult[] results = null;
46        try {
47            results = client.submitMessage(message);
48        } catch (Exception e) {
49            System.err.println("Failed to communicate with the Nexmo Client");
50            e.printStackTrace();
51            throw new RuntimeException("Failed to communicate with the Nexmo Client");
52        }
53
54        // Evaluate the results of the submission attempt ...
55        System.out.println("... Message submitted in [ " + results.length + " ] parts");
56        for (int i=0;i<results.length;i++) {
57            System.out.println("--------- part [ " + (i + 1) + " ] ------------");
58            System.out.println("Status [ " + results[i].getStatus() + " ] ...");
59            if (results[i].getStatus() == SmsSubmissionResult.STATUS_OK)
60                System.out.println("SUCCESS");
61            else if (results[i].getTemporaryError())
62                System.out.println("TEMPORARY FAILURE - PLEASE RETRY");
63            else
64                System.out.println("SUBMISSION FAILED!");
65            System.out.println("Message-Id [ " + results[i].getMessageId() + " ] ...");
66            System.out.println("Error-Text [ " + results[i].getErrorText() + " ] ...");
67
68            if (results[i].getMessagePrice() != null)
69                System.out.println("Message-Price [ " + results[i].getMessagePrice() + " ] ...");
70            if (results[i].getRemainingBalance() != null)
71                System.out.println("Remaining-Balance [ " + results[i].getRemainingBalance() + " ] ...");
72        }
73    }
74
75}
