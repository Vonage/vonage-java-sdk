37/**
38 * NexmoSmsClient.java<br><br>
39 *
40 * Client for talking to the Nexmo REST interface<br><br>
41 *
42 * Usage
43 *
44 * To submit a message, first you should instanciate a NexmoSmsClient, passing the credentials for your Nexmo account on the constructor.
45 * Then, you should instanciate the appropriate {@link com.nexmo.messaging.sdk.messages.Message} subclass depending on which type of message you are going to submit.<br>
46 * The following subclasses are available ...<br><br>
47 *
48 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.TextMessage}<br>
49 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.BinaryMessage}<br>
50 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.WapPushMessage}<br>
51 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.UnicodeMessage}<br>
52 *
53 * Each of these subclasses requires different message parameters to be passed on the constructor.
54 * See the included javadocs for further details.<br><br>
55 *
56 * Once you have a {@link com.nexmo.messaging.sdk.messages.Message} object, you simply pass this to the {@link #submitMessage(Message)} method in the NexmoSmsClient instance.
57 * This will construct and post the request to the Nexmo REST service.<br>
58 * This method will return an array of {@link com.nexmo.messaging.sdk.SmsSubmissionResult}, with 1 entry for every sms message that was sent.<br>
59 * Certain messages, for example, long text messages greater than 160 characters, will require multiple sms's to be submitted.<br>
60 * Each entry in this array will contain an individual messageId as well as an individual status detailing the success or reason for failure of each message.<br><br>
61 *
62 * The list of possible status codes is listed below ..<br><br>
63 *
64 * // 0 = success  -- the message was successfully accepted for delivery by nexmo<br>
65 * // 1 = throttled -- You have exceeded the submission capacity allowed on this account, please back-off and re-try<br>
66 * // 2 = missing params -- Your request is incomplete and missing some mandatory parameters<br>
67 * // 3 = invalid params -- The value of 1 or more parameters is invalid<br>
68 * // 4 = invalid credentials -- The api key / secret you supplied is either invalid or disabled<br>
69 * // 5 = internal error -- An error has occurred in the nexmo platform whilst processing this message<br>
70 * // 6 = invalid message -- The Nexmo platform was unable to process this message, for example, an un-recognized number prefix<br>
71 * // 7 = number barred -- The number you are trying to submit to is blacklisted and may not receive messages<br>
72 * // 8 = partner account barred -- The api key you supplied is for an account that has been barred from submitting messages<br>
73 * // 9 = partner quota exceeded -- Your pre-pay account does not have sufficient credit to process this message<br>
74 * // 10 = too many existing binds -- The number of simultaneous connections to the platform exceeds the capabilities of your account<br>
75 * // 11 = account not enabled for http -- This account is not provisioned for http / rest submission, you should use SMPP instead<br>
76 * // 12 = message too long -- The message length exceeds the maximum allowed<br>
77 * // 13 = comms failure -- There was a network failure attempting to contact Nexmo
78 * // 14 = Invalid Signature -- The signature supplied with this request was not verified successfully
79 * // 15 = invalid sender address -- The sender address was not allowed for this message
80 * // 16 = invalid TTL -- The ttl parameter values, or combination of parameters is invalid
81 * // 17 = number unreachable -- this destination cannot be delivered to at this time (if reachable=true is specified)
82 * // 18 = too many destinations -- There are more than the maximum allowed number of destinations in this request
83 * // 19 = Facility Not Allowed - Your request makes use of a facility that is not enabled on your account

76     * @param validityPeriod The validity period (Time-To-Live) for this message. Specifies the time before this mesage will be expired if not yet delivered
77     * @param networkCode (Optional) use this parameter to force this message to be associated with and delivered on this network. Use this in cases where you want to over-ride
78     *                               the automatic network detection provided by Nexmo. This value will be used in order to determine the pricing and routing for this message.<br>
79     *                               (Note) This feature must be enabled and available on your account or else this value will be ignored.
80     * @param performReachabilityCheck Flag to indicate wether a reachability check should be performed on this message before delivery is attempted. If the destination is
81     *                                 not reachable, the message will be rejected and a reachability status will be returned in the response field smsSubmissionReachabilityStatus<br>
82     *                                 (Note) This feature must be enabled and available on your account or else the message request will be rejected. There may be additional cost
83     *                                 associated with the use of this feature.
84     *
85     * @return SmsSubmissionResult[] an array of results, 1 object for each sms message that was required to submit this message in its entirety
86     *
87     * @throws Exception There has been a general failure either within the Client class, or whilst attempting to communicate with the Nexmo service (eg, Network failure)
88     */
89    public SmsSubmissionResult[] submitMessage(Message message,
90                                               ValidityPeriod validityPeriod,
91                                               String networkCode,
92                                               boolean performReachabilityCheck) throws Exception {
93
94        log.debug("HTTP-SMS-Submission Client .. from [ " + message.getFrom() + " ] to [ " + message.getTo() + " ] msg [ " + message.getMessageBody() + " ] ");
95
96        // From the Message object supplied, construct an appropriate request to be submitted to the Nexmo REST Service.
97
98        // Determine the type parameter based on the type of Message object.
99        boolean binary = message.getType() == Message.MESSAGE_TYPE_BINARY;
00        boolean unicode = message.isUnicode();
01        boolean wapPush = message.getType() == Message.MESSAGE_TYPE_WAPPUSH;
02
03        String mode = "text";
04
05        if (binary)
06            mode = "binary";
07
08        if (unicode)
09            mode = "unicode";
10
11        if (wapPush)
12            mode = "wappush";
13
14        // Construct a query string as a list of NameValuePairs
15
16        List<NameValuePair> params = new ArrayList<NameValuePair>();
17
18        boolean doPost = false;
19
20        params.add(new BasicNameValuePair("api_key", this.apiKey));
21        if (!this.signRequests)
22            params.add(new BasicNameValuePair("api_secret", this.apiSecret));
23        params.add(new BasicNameValuePair("from", message.getFrom()));
24        params.add(new BasicNameValuePair("to", message.getTo()));
25        params.add(new BasicNameValuePair("type", mode));
26        if (wapPush) {
27            params.add(new BasicNameValuePair("url", message.getWapPushUrl()));
28            params.add(new BasicNameValuePair("title", message.getWapPushTitle()));
29            if (message.getWapPushValidity() > 0)
30                params.add(new BasicNameValuePair("validity", "" + message.getWapPushValidity()));
31        } else if (binary) {
32            // Binary Message
33            if (message.getBinaryMessageUdh() != null)
34                params.add(new BasicNameValuePair("udh", HexUtil.bytesToHex(message.getBinaryMessageUdh())));
35            params.add(new BasicNameValuePair("body", HexUtil.bytesToHex(message.getBinaryMessageBody())));
36        } else {
37            // Text Message
38            params.add(new BasicNameValuePair("text", message.getMessageBody()));
39            if (message.getMessageBody() != null && message.getMessageBody().length() > 255)
40                doPost = true;
41        }
42
43        if (message.getClientReference() != null)
44            params.add(new BasicNameValuePair("client-ref", message.getClientReference()));
45
46        params.add(new BasicNameValuePair("status-report-req", "" + message.getStatusReportRequired()));
47
48        if (message.getMessageClass() != null)
49            params.add(new BasicNameValuePair("message-class", "" + message.getMessageClass().getMessageClass()));
50
51        if (message.getProtocolId() != null)
52            params.add(new BasicNameValuePair("protocol-id", "" + message.getProtocolId()));
53
54        if (validityPeriod != null) {
55            if (validityPeriod.getTimeToLive() != null)
56                params.add(new BasicNameValuePair("ttl", "" + validityPeriod.getTimeToLive().intValue()));
57            if (validityPeriod.getValidityPeriodHours() != null)
58                params.add(new BasicNameValuePair("ttl-hours", "" + validityPeriod.getValidityPeriodHours().intValue()));
59            if (validityPeriod.getValidityPeriodMinutes() != null)
60                params.add(new BasicNameValuePair("ttl-minutes", "" + validityPeriod.getValidityPeriodMinutes().intValue()));
61            if (validityPeriod.getValidityPeriodSeconds() != null)
62                params.add(new BasicNameValuePair("ttl-seconds", "" + validityPeriod.getValidityPeriodSeconds().intValue()));
63        }
64
65        if (networkCode != null)
66            params.add(new BasicNameValuePair("network-code", networkCode));
67
68        if (performReachabilityCheck)
69            params.add(new BasicNameValuePair("test-reachable", "true"));
70
71        if (this.signRequests)
72            RequestSigning.constructSignatureForRequestParameters(params, this.signatureSecretKey);
73
74        String baseUrl = this.useSSL ? this.baseUrlHttps : this.baseUrlHttp;
75
76        // Now that we have generated a query string, we can instanciate a HttpClient,
77        // construct a POST or GET method and execute to submit the request
78        String response = null;
79        for (int pass=1;pass<=2;pass++) {
80            HttpUriRequest method = null;
81            doPost = true;
82            String url = null;
83            if (doPost) {
84                HttpPost httpPost = new HttpPost(baseUrl);
85                httpPost.setEntity(new UrlEncodedFormEntity(params, "UTF-8")); 
86                method = httpPost;
87                url = baseUrl + "?" + URLEncodedUtils.format(params, "utf-8");
88            } else {
89                String query = URLEncodedUtils.format(params, "utf-8");
90                method = new HttpGet(baseUrl + "?" + query);
91                url = method.getRequestLine().getUri();
92            }
93            
94            try {
95                if (this.httpClient == null)
96                    this.httpClient = HttpClientUtils.getInstance(this.connectionTimeout, this.soTimeout).getNewHttpClient();
97                HttpResponse httpResponse = this.httpClient.execute(method);
98                int status = httpResponse.getStatusLine().getStatusCode();
99                if (status != 200)
00                    throw new Exception("got a non-200 response [ " + status + " ] from Nexmo-HTTP for url [ " + url + " ] ");
01                response = new BasicResponseHandler().handleResponse(httpResponse);
02                log.info(".. SUBMITTED NEXMO-HTTP URL [ " + url + " ] -- response [ " + response + " ] ");
03                break;
04            } catch (Exception e) {
05                method.abort();
06                log.info("communication failure: " + e);
07                String exceptionMsg = e.getMessage();
08                if (exceptionMsg.indexOf("Read timed out") >= 0) {
09                    log.info("we're still connected, but the target did not respond in a timely manner ..  drop ...");
10                } else {
11                    if (pass == 1) {
12                        log.info("... re-establish http client ...");
13                        this.httpClient = null;
14                        continue;
15                    }
16                }
17
18                // return a COMMS failure ...
19                SmsSubmissionResult[] results = new SmsSubmissionResult[1];
20                results[0] = new SmsSubmissionResult(SmsSubmissionResult.STATUS_COMMS_FAILURE,
21                                                     null,
22                                                     null,
23                                                     "Failed to communicate with NEXMO-HTTP url [ " + url + " ] ..." + e,
24                                                     message.getClientReference(),
25                                                     null,
26                                                     null,
27                                                     true,
28                                                     null);
29                return results;
30            }
31        }
32
33        // parse the response doc ...
34
35        /*
36            We receive a response from the api that looks like this, parse the document
37            and turn it into an array of SmsSubmissionResult, one object per <message> node
38
39                <mt-submission-response>
40                    <messages count='x'>
41                        <message>
42                            <to>xxx</to>
43                            <messageId>xxx</messageId>
44                            <status>xx</status>
45                            <errorText>ff</errorText>
46                            <clientRef>xxx</clientRef>
47                            <remainingBalance>##.##</remainingBalance>
48                            <messagePrice>##.##</messagePrice>
49                            <reachability status='x' description='xxx' />
50                        </message>
51                    </messages>
52                </mt-submission-response>
53        */
54
55        List<SmsSubmissionResult> results = new ArrayList<SmsSubmissionResult>();
56
57        Document doc = null;
58        synchronized(this.documentBuilder) {
59            try {
60                doc = this.documentBuilder.parse(new InputSource(new StringReader(response)));
61            } catch (Exception e) {
62                throw new Exception("Failed to build a DOM doc for the xml document [ " + response + " ] ", e);
63            }
64        }
65
66        NodeList replies = doc.getElementsByTagName("mt-submission-response");
67        for (int i=0;i<replies.getLength();i++) {
68            Node reply = replies.item(i);
69            NodeList messageLists = reply.getChildNodes();
70            for (int i2=0;i2<messageLists.getLength();i2++) {
71                Node messagesNode = messageLists.item(i2);
72                if (messagesNode.getNodeType() != Node.ELEMENT_NODE)
73                    continue;
74                if (messagesNode.getNodeName().equals("messages")) {
75                    NodeList messages = messagesNode.getChildNodes();
76                    for (int i3=0;i3<messages.getLength();i3++) {
77                        Node messageNode = messages.item(i3);
78                        if (messageNode.getNodeType() != Node.ELEMENT_NODE)
79                            continue;
80
81                        int status = -1;
82                        String messageId = null;
83                        String destination = null;
84                        String errorText = null;
85                        String clientReference = null;
86                        BigDecimal remainingBalance = null;
87                        BigDecimal messagePrice = null;
88                        SmsSubmissionReachabilityStatus smsSubmissionReachabilityStatus = null;
89
90                        NodeList nodes = messageNode.getChildNodes();
91                        for (int i4=0;i4<nodes.getLength();i4++) {
92                            Node node = nodes.item(i4);
93                            if (node.getNodeType() != Node.ELEMENT_NODE)
94                                continue;
95                            if (node.getNodeName().equals("messageId")) {
96                                messageId = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
97                            } else if (node.getNodeName().equals("to")) {
98                                destination = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
99                            } else if (node.getNodeName().equals("status")) {
00                                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
01                                try {
02                                    status = Integer.parseInt(str);
03                                } catch (NumberFormatException e) {
04                                    log.error("xml parser .. invalid value in <status> node [ " + str + " ] ");
05                                    status = SmsSubmissionResult.STATUS_INTERNAL_ERROR;
06                                }
07                            } else if (node.getNodeName().equals("errorText")) {
08                                errorText = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
09                            } else if (node.getNodeName().equals("clientRef")) {
10                                clientReference = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
11                            } else if (node.getNodeName().equals("remainingBalance")) {
12                                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
13                                try {
14                                    if (str != null)
15                                        remainingBalance = new BigDecimal(str);
16                                } catch (NumberFormatException e) {
17                                    log.error("xml parser .. invalid value in <remainingBalance> node [ " + str + " ] ");
18                                }
19                            } else if (node.getNodeName().equals("messagePrice")) {
20                                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
21                                try {
22                                    if (str != null)
23                                        messagePrice = new BigDecimal(str);
24                                } catch (NumberFormatException e) {
25                                    log.error("xml parser .. invalid value in <messagePrice> node [ " + str + " ] ");
26                                }
27                            } else if (node.getNodeName().equals("reachability")) {
28                                NamedNodeMap attributes = node.getAttributes();
29                                Node attr = attributes.getNamedItem("status");
30                                String str = attr == null ? "" + SmsSubmissionReachabilityStatus.REACHABILITY_STATUS_UNKNOWN : attr.getNodeValue();
31                                int reachabilityStatus = SmsSubmissionReachabilityStatus.REACHABILITY_STATUS_UNKNOWN;
32                                try {
33                                    reachabilityStatus = Integer.parseInt(str);
34                                } catch (NumberFormatException e) {
35                                    log.error("xml parser .. invalid value in 'status' attribute in <reachability> node [ " + str + " ] ");
36                                    reachabilityStatus = SmsSubmissionReachabilityStatus.REACHABILITY_STATUS_UNKNOWN;
37                                }
38
39                                attr = attributes.getNamedItem("description");
40                                String description = attr == null ? "-UNKNOWN-" : attr.getNodeValue();
41
42                                smsSubmissionReachabilityStatus = new SmsSubmissionReachabilityStatus(reachabilityStatus, description);
43                            } else if (node.getNodeName().equals("network")) {
44                                // placeholder for future functionality
45                            } else
46                                log.error("xml parser .. unknown node found in status-return, expected [ messageId, to, status, errorText, clientRef, messagePrice, remainingBalance, reachability ] -- found [ " + node.getNodeName() + " ] ");
47                        }
48
49                        if (status == -1)
50                            throw new Exception("Xml Parser - did not find a <status> node");
51
52                        // Is this a temporary error ?
53                        boolean temporaryError = (status == SmsSubmissionResult.STATUS_THROTTLED ||
54                                                  status == SmsSubmissionResult.STATUS_INTERNAL_ERROR ||
55                                                  status == SmsSubmissionResult.STATUS_TOO_MANY_BINDS);
56
57                        results.add(new SmsSubmissionResult(status,
58                                                            destination,
59                                                            messageId,
60                                                            errorText,
61                                                            clientReference,
62                                                            remainingBalance,
63                                                            messagePrice,
64                                                            temporaryError,
65                                                            smsSubmissionReachabilityStatus));
66                    }
67                }
68            }
69        }
70
71        return results.toArray(new SmsSubmissionResult[0]);
72    }
